public abstract class BasePresenterImpl<V extends BaseView> implements BasePresenter<V> {
    private V mView;

    @Override
    public void attachView(V view) {
        mView = view;
    }

    @Override
    public void detachView() {
        mView = null;
    }

    public V getView() {
        if (mView == null) {
            return setView();
        } else {
            return mView;
        }
    }

    public abstract V setView();
}
